Multi-protocol router for agentic commerce:

```javascript
const protocolRouter = {
  visa: require('@visa/trusted-agent-protocol'),
  mastercard: require('@mastercard/agent-pay'),
  paypal: require('@paypal/agent-ready'),
  stripe: require('stripe'),
  x402: require('@x402/client')
};

async function routeAgentPayment(context) {
  // Determine optimal protocol based on transaction context
  const protocol = selectProtocol(context);

  switch (protocol) {
    case 'visa-tap':
      // Physical goods through card networks
      return await protocolRouter.visa.processWithTAP({
        amount: context.amount,
        agentId: context.agentId,
        intent: context.intent
      });

    case 'stripe-acp':
      // ChatGPT/AI platform checkout
      return await protocolRouter.stripe.paymentIntents.create({
        amount: context.amount * 100,
        currency: 'usd',
        metadata: { acp_version: '1.0', agent_id: context.agentId }
      });

    case 'x402':
      // Machine-to-machine, API calls
      const client = new protocolRouter.x402({
        wallet: process.env.AGENT_WALLET,
        preferences: { preferredChain: 'base' }
      });
      return await client.fetch(context.endpoint);

    default:
      throw new Error(`Unknown protocol: ${protocol}`);
  }
}

function selectProtocol(context) {
  if (context.type === 'physical_goods') {
    return context.cardNetwork === 'visa' ? 'visa-tap' : 'mastercard-agent-pay';
  }
  if (context.platform === 'chatgpt' || context.platform === 'copilot') {
    return 'stripe-acp';
  }
  if (context.type === 'api_call' || context.settlement === 'instant') {
    return 'x402';
  }
  return 'paypal-agent-ready';  // Default fallback
}

// Usage: routeAgentPayment({ type: 'api_call', amount: 0.01, agentId: 'data-agent' });
```

```javascript
const x402 = require('@x402/client');
const usdc = require('@circle/usdc-sdk');

class AgenticMicropaymentHandler {
  constructor(config) {
    this.x402Client = new x402.Client({
      wallet: config.walletAddress,
      chain: config.preferredChain || 'base'
    });
    this.usdcClient = new usdc.Client({
      apiKey: config.circleApiKey
    });
    this.transactionBuffer = [];
    this.batchThreshold = 100;  // Batch small transactions
  }

  async payForAPICall(endpoint, expectedCost) {
    // Micropayment: use x402 for instant, low-cost settlement
    if (expectedCost < 0.10) {
      return await this.x402Client.fetch(endpoint, {
        maxPayment: expectedCost.toString()
      });
    }

    // Larger payments: use USDC with batching
    this.transactionBuffer.push({
      endpoint,
      cost: expectedCost,
      timestamp: Date.now()
    });

    if (this.transactionBuffer.length >= this.batchThreshold) {
      await this.settleBatch();
    }

    return { status: 'queued', batchId: this.currentBatchId };
  }

  async settleBatch() {
    const totalAmount = this.transactionBuffer.reduce(
      (sum, tx) => sum + tx.cost, 0
    );

    const settlement = await this.usdcClient.transfer({
      amount: totalAmount,
      currency: 'USDC',
      destination: process.env.SETTLEMENT_ADDRESS,
      memo: `Batch ${this.currentBatchId}: ${this.transactionBuffer.length} transactions`
    });

    this.transactionBuffer = [];
    return settlement;
  }

  // Price-per-crawl pattern mentioned in American Banker
  async crawlWithPayment(apiEndpoint, pricePerCrawl = 0.005) {
    const result = await this.payForAPICall(apiEndpoint, pricePerCrawl);
    return result;
  }
}

// Usage:

  walletAddress: process.env.AGENT_WALLET,
  circleApiKey: process.env.CIRCLE_API_KEY,
  preferredChain: 'base'
```
